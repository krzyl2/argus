<!-- openssl cert generation, RESEARCH "openssl mTLS Cert Generation" code block -->
CA:        openssl genrsa -out ca.key 4096; openssl req -new -x509 -key ca.key -out ca.crt -days 730 -subj "/CN=ArgusCA"
Server SAN: subjectAltName=IP:<GPU_HOST_IP>,DNS:<GPU_HOST_NAME>   (CRITICAL — both required, PITFALL 2)
Server:    openssl req -new -key server.key -subj "/CN=<GPU_HOST_NAME>" → x509 -req -extfile server-ext.cnf -days 730
Client:    openssl req -new -key client.key -subj "/CN=edge-host" → x509 -req -days 730
</interfaces>
